Windows下程序执行时间测试

#include <time.h>
#include <stdio.h>

int main()
{
  clock_t start;
  clock_t finish;

  start = clock();
  //do some thing
  finish = clock();

  printf("time = %f\r\n", (double(finish - start)/CLOCKS_PER_SEC));
  fflush(stdout);

  return 0;
}
